	public.taxi_zone."Borough"
	, SUM(total_amount) AS total_amount
FROM public.green_taxi_data
LEFT JOIN public.taxi_zone 
ON public.green_taxi_data."PULocationID" = public.taxi_zone."LocationID"
GROUP BY public.taxi_zone."Borough"
ORDER BY SUM(total_amount) DESC
LIMIT 10
